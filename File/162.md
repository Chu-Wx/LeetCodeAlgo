```python
def findPeakElement(self, nums: List[int]) -> int:  
    nums.append(-float("inf")) # top tail both are -inf
        lo, hi = 0, len(nums) - 1
        while lo < hi:
            mid = (lo + hi) // 2
            if nums[mid] < nums[mid + 1]: # if left high means peak on left then go left
                lo = mid + 1
            else: # peak on right then go right 
                hi = mid
        return lo
```
